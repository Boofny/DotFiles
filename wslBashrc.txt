# If not running interactively, don't do anything
[[ $- != *i* ]] && return

alias ..='cd ..'
alias c='clear'
alias cls='clear'
alias ls='lsd'
alias lsa='lsd -a'
alias lst='lsd --tree'
alias lsl='lsd -l'
alias vi='nvim'
alias neo='fastfetch'
alias grep='grep --color=auto'
alias postStatus='sudo systemctl status postgresql'
alias postStart='sudo systemctl start postgresql'
alias postStop='sudo systemctl stop postgresql'
alias postUser='pgcli -U myuser -d mydb -h localhost'

alias dockStart='sudo systemctl start docker.service'
alias dockStop='sudo systemctl stop docker.service'
alias dockStat='sudo systemctl status docker.service'

PS1='[\u@\h \W]\$ '
eval "$(starship init bash)"
export EDITOR=nvim

function y() {
	local tmp="$(mktemp -t "yazi-cwd.XXXXXX")" cwd
	yazi "$@" --cwd-file="$tmp"
	IFS= read -r -d '' cwd < "$tmp"
	[ -n "$cwd" ] && [ "$cwd" != "$PWD" ] && builtin cd -- "$cwd"
	rm -f -- "$tmp"
}
export PATH=$PATH:$(go env GOPATH)/bin
